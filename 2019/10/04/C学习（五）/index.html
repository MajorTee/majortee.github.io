结构体作为函数参数

void copy(Teacher from, Teacher to)
{
    to = from;
}
copy(t1, t2);
//此时只是把t1给了form， t2给了to，跟原本的t1，t2没关系

void copy(Teacher* from, Teacher* to)
{
    *to = *from;
}
copy(&t1, &t2);
//这样才可以把结构体内 内容拷贝

结构体的深浅拷贝

struct Teacher
{
    char name[64];
    char* otherName;//需要自己分配内存
    int age;
}Teacher;

void copy(Teacher* from, Teacher* to)
{
    *to = *from;//只是个浅拷贝
    //memcpy(to, form, sizeof(Teacher)) // 一样
}

t1.otherName = (char*)malloc(100);
copy(&t1, &t2);
//*otherName拷贝出现问题
// t1.otherName t2.otherName是同一个地址值，但是指向的内存空间并没有拷贝  《== 浅拷贝

//解决
void copy(Teacher* from, Teacher* to)
{
    *to = *from;//只是个浅拷贝
    //memcpy(to, form, sizeof(Teacher)) // 一样

    to.otherName = (char*)malloc(100);
    strcpy(to.otherName, from.otherName);
}
